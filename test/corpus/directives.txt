================================================================================
If expression spread between multiple directives
================================================================================

<%= if true do %>
  <%= @x %>
<% end %>

--------------------------------------------------------------------------------

(fragment
  (directive_group
    (directive
      (partial_expression_value))
    (directive
      (expression_value))
    (directive
      (ending_expression_value))))

================================================================================
Case expression spread between multiple directives
================================================================================

<%= case @x do %>
  <%= ^x -> %>X, <%= x %>
  <%= _ -> %>Not X
<% end %>

--------------------------------------------------------------------------------

(fragment
  (directive_group
    (directive
      (partial_expression_value))
    (directive
      (partial_expression_value))
    (text)
    (directive
      (expression_value))
    (directive
      (partial_expression_value))
    (text)
    (directive
      (ending_expression_value))))

================================================================================
Nested if expressions

  <% end %>
