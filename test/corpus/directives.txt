================================================================================
If expression spread between multiple directives
================================================================================

<%= if true do %>
  <%= @x %>
<% end %>

--------------------------------------------------------------------------------

(fragment
  (directive
    (partial_expression_value))
  (directive
    (expression_value))
  (directive
    (ending_expression_value)))

================================================================================
Case expression spread between multiple directives
================================================================================

<%= case @x do %>
  <%= ^x -> %>X, <%= x %>
  <%= _ -> %>Not X
<% end %>

--------------------------------------------------------------------------------

(fragment
  (directive
    (partial_expression_value))
  (directive
    (partial_expression_value))
  (text)
  (directive
    (expression_value))
  (directive
    (partial_expression_value))
  (text)
  (directive
    (ending_expression_value)))

================================================================================
Curly braces as directive
================================================================================

<div>
  { @message }
  {@message}
  {"#{1}"}
</div>

--------------------------------------------------------------------------------

(fragment
  (tag
    (start_tag
      (tag_name))
    (directive
      (expression_value))
    (directive
      (expression_value))
    (directive
      (expression_value))
    (end_tag
      (tag_name))))
